function goToIndex() { location.replace('index.html'); } function onCacheReady() { setMessage('Cache Installed Successfully'); localStorage.setItem('cacheInstalled', 'true'); setTimeout(goToIndex, 1000); } function onError() { setMessage('Offline cache installation failed'); } window.addEventListener('load', function () { if (!window.applicationCache) { setMessage('Application cache is not supported'); return; } var appCache = window.applicationCache; appCache.addEventListener('progress', onProgress, false); appCache.addEventListener('cached', onCacheReady, false); appCache.addEventListener('updateready', onCacheReady, false); appCache.addEventListener('noupdate', onCacheReady, false); appCache.addEventListener('error', onError, false); }, false);
